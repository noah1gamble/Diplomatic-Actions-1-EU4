

enforce_vassalage = {
    category = great_powers     
    
    alert_index = 3          
    
    require_acceptance = no
    
    is_visible = {
		ROOT = {
			is_great_power = yes
		}
		FROM = {
			vassal_of = ROOT
		}
    }
	
    is_allowed = {
		AND = {
			FROM = {
				vassal_of = ROOT
			}	
			num_of_vassals = 1
			num_of_free_diplomatic_relations = 0
		}
		ROOT = { 
			NOT = { 
				has_country_modifier = da1_vassalage_simple_policies 
				has_country_modifier = da1_vassalage_moderate_policies
			}
		is_great_power = yes	
		}
	}
	
	alert_tooltip = enforce_vassalage_tooltip
	
    on_accept = {
			ROOT = {
				country_event = { 
					id = da1.2
				}				
			}
		}		
	
	}
	
enforce_subject = {
    category = great_powers     
    
    alert_index = 3          
    
    require_acceptance = no
    
    is_visible = {
		ROOT = {
			is_great_power = yes
		}
		FROM = {
			junior_union_with = ROOT
		}		
    }
	
    is_allowed = {
		AND = {
			FROM = {
					junior_union_with = ROOT
				}
			num_of_subjects = 1
			num_of_free_diplomatic_relations = 0
			}
		ROOT = { 
			NOT = { 
				has_country_modifier = da1_subject_simple_policies 
				has_country_modifier = da1_subject_moderate_policies
				has_country_modifier = da1_subject_aggressive_policies
			}
		is_great_power = yes	
		}
	}
	
	alert_tooltip = enforce_subject_tooltip
	
    on_accept = {
			ROOT = {
				country_event = { 
					id = da1.1
				}				
			}
		}
	}
